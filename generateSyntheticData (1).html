    import matplotlib.pyplot as plt, seaborn as sns
import matplotlib.style as style
def correlation_matrix(df, threshold = 0.8):
    corr = df.corr()
    mask = np.zeros_like(corr, dtype = np.bool)
    mask[np.triu_indices_from(mask)] = True
    f, ax = plt.subplots(figsize = (15, 7.2))
    cmap = sns.diverging_palette(220, 10, as_cmap = True)
    sns.heatmap(corr, mask = mask, cmap = cmap, square = True, linewidths = .5, cbar_kws = {"shrink": .5})#, annot = True)
    ax.set_title('Correlation Matrix of Data')

    def generateScore(bmi,water,alc,grains,fat,dairy,sweets,fried,vegetables,cardio,cardioIntensity,
                  muscleTraining,sittingHours,sleepingHours,smoke,tobacco,worklifeBalance,stress,exerciseTime,steps):
    score=0
    if bmi < 18:
        score = score + 20
    else:
        score = score + (60-bmi)
    
    if water == "6-9 glass":
        score = score + 10
    elif water == "10+ glass":
        score = score + 20
    
    if alc == "1-4 servings":
        score = score - 10
    elif alc == "4+ servings":
        score = score - 20
        
    if grains == "Yes":
        score = score + 20
    elif grains == "Sometimes":
        score = score + 10
    
    if fat == "Yes":
        score = score + 20
    elif fat == "Sometimes":
        score = score + 10
        
    if dairy == "Yes":
        score = score + 20
    elif dairy == "Sometimes":
        score = score + 10
        
    if sweets == "Yes":
        score = score + 20
    elif sweets == "Sometimes":
        score = score + 10
        
    if fried == "Rarely":
        score = score + 20
    elif fried == "Mostly":
        score = score + 10
        
    if vegetables == "Yes":
        score = score + 20
    elif vegetables == "Sometimes":
        score = score + 10
        
    if cardio == "More than 3 times a week":
        score = score + 40
    elif cardio == "1-2 times week":
        score = score + 20
    
    if cardioIntensity == "Very Hard":
        score = score + 40
    elif cardioIntensity == "Hard":
        score = score + 20
    elif cardioIntensity == "Light":
        score = score + 10
    
    if muscleTraining == "Very Hard":
        score = score + 40
    elif muscleTraining == "Hard":
        score = score + 20
    elif muscleTraining == "Light":
        score = score + 10
        
    if sittingHours < 5:
        score = score + 40
    elif sittingHours <7: 
        score = score + 20
    elif sittingHours < 13:
        score = score + 10
    
    if sleepingHours < 3:
        score = score + 0
    elif sittingHours < 5: 
        score = score + 10
    elif sittingHours < 7: 
        score = score + 20
    else:
        score = score + 40
    
    if smoke == "1-4 cigrattes":
        score = score - 10
    elif smoke == "15+ cigrattes":
        score = score - 20
    
    if tobacco == "Yes":
        score = score - 20
    
    if worklifeBalance == "Yes":
        score = score + 20
    elif worklifeBalance == "Sometimes":
        score = score + 10
    
    if stress == "Yes":
        score = score + 20
    elif stress == "Sometimes":
        score = score + 10
          
    if exerciseTime < 0.1 :
        score = score + 0
    elif exerciseTime < 0.5:
        score = score + 10
    elif exerciseTime > 0.5:
        score = score + 20
    
    if steps < 50031 :
        score = score + 0
    elif steps < 178757:
        score = score + 10
    elif steps < 239571:
        score = score + 20
    elif steps > 239571:
        score = score + 40
    
    return ((score/420)*100)
